from PIL import Image, ImageDraw
import time

def make_test_image(text, filename):
    img = Image.new("RGB", (600, 80), "white")
    draw = ImageDraw.Draw(img)
    draw.text((10, 20), text, fill="black")
    img.save(filename)
    return img

img_ar    = make_test_image("مرحبا بك في نظام OCR العربي", "/tmp/test_ar.png")
img_en    = make_test_image("OmniFile AI Processor v5.0", "/tmp/test_en.png")
img_mixed = make_test_image("Test Arabic OCR نص تجريبي", "/tmp/test_mixed.png")

results = {}

# EasyOCR
try:
    import easyocr
    t0 = time.time()
    reader = easyocr.Reader(["ar","en"], gpu=USE_GPU, verbose=False)
    res = reader.readtext("/tmp/test_mixed.png")
    text = " ".join(x[1] for x in res)
    conf = sum(x[2] for x in res)/len(res) if res else 0
    elapsed = time.time() - t0
    results["EasyOCR"] = {"text": text, "conf": conf, "time": elapsed}
    print(f"EasyOCR  : [{conf:.0%}] {text[:55]} ({elapsed:.1f}s)")
except Exception as e:
    print(f"EasyOCR  : FAILED -> {e}")
    results["EasyOCR"] = {"text": "", "error": str(e)}

# Tesseract
try:
    import pytesseract
    t0 = time.time()
    text = pytesseract.image_to_string(img_mixed, lang="ara+eng").strip()
    elapsed = time.time() - t0
    results["Tesseract"] = {"text": text, "time": elapsed}
    print(f"Tesseract: {text[:55]} ({elapsed:.1f}s)")
except Exception as e:
    print(f"Tesseract: FAILED -> {e}")
    results["Tesseract"] = {"text": "", "error": str(e)}

# TrOCR (only in balanced/high profile)
if profile_cfg["enable_trocr"]:
    try:
        from transformers import TrOCRProcessor, VisionEncoderDecoderModel
        t0 = time.time()
        proc  = TrOCRProcessor.from_pretrained("microsoft/trocr-base-printed")
        model = VisionEncoderDecoderModel.from_pretrained("microsoft/trocr-base-printed")
        pixel_values = proc(images=img_en, return_tensors="pt").pixel_values
        ids  = model.generate(pixel_values)
        text = proc.batch_decode(ids, skip_special_tokens=True)[0]
        elapsed = time.time() - t0
        results["TrOCR"] = {"text": text, "time": elapsed}
        print(f"TrOCR    : {text[:55]} ({elapsed:.1f}s)")
    except Exception as e:
        print(f"TrOCR    : FAILED -> {e}")
        results["TrOCR"] = {"text": "", "error": str(e)}
else:
    print(f"TrOCR    : SKIPPED (profile={PROFILE})")

## 16 — الموجّه الذكي للمحركات (اقتراح Claude)

> بدلاً من تشغيل كل المحركات معاً، يختار المحركين الأمثل حسب السياق — يوفر 60% من الذاكرة.


class EngineRouter:
    """
    الموجّه الذكي للمحركات.
    اقتراح من مراجعة Claude المعمارية: استبدال التنفيذ المتوازي بـ Router يختار 2 محركات فقط.
    """

    def __init__(self, profile="balanced", use_gpu=False):
        self.profile = profile
        self.use_gpu = use_gpu

    def select_engines(self, image_quality=0.8, language="ar", block_type="paragraph"):
        recommendations, reasons = [], []

        if self.profile == "low":
            return ["Tesseract"], ["low-end profile — single engine"]

        # Arabic or mixed -> EasyOCR (best for Arabic)
        if language in ["ar", "mixed"]:
            recommendations.append("EasyOCR")
            reasons.append("Arabic/mixed detected")

        # Handwriting -> TrOCR
        if block_type == "handwriting" and image_quality > 0.7:
            recommendations.append("TrOCR")
            reasons.append("handwriting block")

        # Low quality -> Tesseract (tolerant to noise)
        if image_quality < 0.6:
            if "Tesseract" not in recommendations:
                recommendations.append("Tesseract")
            reasons.append("low image quality")

        # English with good quality -> TrOCR
        if language == "en" and image_quality >= 0.75 and "TrOCR" not in recommendations:
            recommendations.append("TrOCR")
            reasons.append("English + good quality")

        # Default fallback
        if not recommendations:
            recommendations = ["EasyOCR", "Tesseract"]
            reasons = ["default fallback"]

        # Remove duplicates, keep max 2
        seen = []
        for e in recommendations:
            if e not in seen:
                seen.append(e)
        return seen[:2], reasons

def run_ocr_ui(image, use_easy, use_tess, langs):

```python
from pathlib import Path
from modules.nlp.arabic_nlp_utils import normalize_for_comparison

class CorrectionLearner:
    """نظام تعلم من تصحيحات المستخدم - يدعم التخزين المحلي والسحابي"""
    
    def __init__(self, db_path="artifacts/corrections.db", sync_github=True, sync_gdrive=False):
        self.db_path = Path(db_path)
        self.db_path.parent.mkdir(exist_ok=True)
        self._init_db()
        self.corrections = self._load_corrections()
        self.sync_github = sync_github
        self.sync_gdrive = sync_gdrive
        
    def _init_db(self):
        """تهيئة قاعدة بيانات SQLite للتخزين المحلي"""
        conn = sqlite3.connect(self.db_path)
        c = conn.cursor()
        c.execute('''
            CREATE TABLE IF NOT EXISTS corrections (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                wrong_text TEXT NOT NULL,
                correct_text TEXT NOT NULL,
                language TEXT,
                context_hash TEXT,
                confidence REAL DEFAULT 0.0,
                usage_count INTEGER DEFAULT 1,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                UNIQUE(wrong_text, language)
            )
        ''')
        c.execute('CREATE INDEX IF NOT EXISTS idx_wrong ON corrections(wrong_text, language)')
        conn.commit()
        conn.close()
    
    def add_correction(self, wrong: str, correct: str, lang: str, 
                      context: str = "", confidence: float = 0.0):
        """إضافة تصحيح جديد مع زيادة عداد الاستخدام"""
        # تطبيع النص للمقارنة
        wrong_norm = normalize_for_comparison(wrong)
        correct_norm = normalize_for_comparison(correct)
        context_hash = hashlib.md5(context.encode()).hexdigest()[:8]
        
        conn = sqlite3.connect(self.db_path)
        c = conn.cursor()
        
        # تحديث إذا كان التصحيح موجوداً
        c.execute('''
            INSERT INTO corrections (wrong_text, correct_text, language, context_hash, confidence)
            VALUES (?, ?, ?, ?, ?)
            ON CONFLICT(wrong_text, language) 
            DO UPDATE SET 
                correct_text = excluded.correct_text,
                usage_count = usage_count + 1,
                confidence = (confidence + excluded.confidence) / 2
        ''', (wrong_norm, correct_norm, lang, context_hash, confidence))
        
        conn.commit()
        conn.close()
        
        # تحديث القاموس في الذاكرة
        key = f"{lang}:{wrong_norm}"
        self.corrections[key] = {
            "correct": correct_norm,
            "count": self.corrections.get(key, {}).get("count", 0) + 1,
            "confidence": confidence
        }
        
        # مزامنة اختيارية مع السحابة
        if self.sync_github:
            self._sync_to_github()
        if self.sync_gdrive:
            self._sync_to_gdrive()
    
    def suggest_correction(self, text: str, lang: str, top_k: int = 3) -> list:
        """اقتراح تصحيحات بناءً على النص المدخل واللغة"""
        text_norm = normalize_for_comparison(text)
        suggestions = []
        
        # بحث مباشر
        key = f"{lang}:{text_norm}"
        if key in self.corrections:
            suggestions.append({
                "text": self.corrections[key]["correct"],
                "confidence": self.corrections[key]["confidence"],
                "source": "exact_match"
            })
        
        # بحث ضبابي (Fuzzy) للكلمات المشابهة
        for k, v in self.corrections.items():
            if k.startswith(f"{lang}:"):
                stored_wrong = k.split(":", 1)[1]
                # حساب التشابه (يمكن استخدام rapidfuzz للأداء الأفضل)
                from modules.nlp.arabic_nlp_utils import arabic_normalized_similarity
                sim = arabic_normalized_similarity(text_norm, stored_wrong)
                if sim > 0.7 and v["correct"] not in [s["text"] for s in suggestions]:
                    suggestions.append({
                        "text": v["correct"],
                        "confidence": sim * v.get("confidence", 0.5),
                        "source": "fuzzy_match"
                    })
        
        # ترتيب حسب الثقة وإرجاع الأفضل
        return sorted(suggestions, key=lambda x: x["confidence"], reverse=True)[:top_k]
    
    def _sync_to_github(self):
```

```python
            corrected_blocks.append(block)
        
        raw_results["blocks"] = corrected_blocks
```

## 📝 الخطوة 1: إنشاء ملف `modules/nlp/correction_learner.py`

```python
# modules/nlp/correction_learner.py
"""
CorrectionLearner: نظام تعلم تفاعلي من تصحيحات المستخدم
يدعم: العربية، الإنجليزية، الألمانية | تخزين محلي + مزامنة سحابية
"""

import json, sqlite3, hashlib, time
from pathlib import Path
from typing import List, Dict, Optional, Tuple
from collections import defaultdict

from modules.nlp.arabic_nlp_utils import normalize_for_comparison, arabic_normalized_similarity

class CorrectionLearner:
    """نظام ذكي يتعلم من تصحيحات المستخدم لتحسين استخراج النصوص"""
    
    MIN_CONFIDENCE_FOR_AUTO = 0.85  # عتبة الثقة للتطبيق التلقائي
    MIN_USAGE_FOR_TRUST = 3         # عدد المرات اللازمة لاعتبار التصحيح موثوقاً
    
    def __init__(self, db_path: str = "artifacts/corrections.db", 
                 sync_enabled: bool = True):
        self.db_path = Path(db_path)
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        self.sync_enabled = sync_enabled
        self._init_database()
        self._memory_cache = {}  # ذاكرة مؤقتة للوصول السريع
        
    def _init_database(self):
        """تهيئة قاعدة بيانات SQLite مع فهارس للأداء"""
        conn = sqlite3.connect(self.db_path)
        c = conn.cursor()
        
        c.execute('''
            CREATE TABLE IF NOT EXISTS corrections (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                wrong_hash TEXT NOT NULL,
                wrong_text TEXT NOT NULL,
                correct_text TEXT NOT NULL,
                language TEXT NOT NULL,
                context_preview TEXT,
                confidence REAL DEFAULT 0.5,
                usage_count INTEGER DEFAULT 1,
                user_feedback_score REAL DEFAULT 1.0,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                UNIQUE(wrong_hash, language)
            )
        ''')
        
        # فهارس لتسريع البحث
        c.execute('CREATE INDEX IF NOT EXISTS idx_wrong_hash ON corrections(wrong_hash)')
        c.execute('CREATE INDEX IF NOT EXISTS idx_lang ON corrections(language)')
        c.execute('CREATE INDEX IF NOT EXISTS idx_confidence ON corrections(confidence DESC)')
        
        # جدول لتتبع تاريخ التعديلات (للتراجع)
        c.execute('''
            CREATE TABLE IF NOT EXISTS correction_history (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                correction_id INTEGER,
                action TEXT,  -- 'created', 'updated', 'deleted', 'undone'
                timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                FOREIGN KEY(correction_id) REFERENCES corrections(id)
            )
        ''')
        
        conn.commit()
        conn.close()
    
    def _hash_text(self, text: str, lang: str) -> str:
        """إنشاء بصمة فريدة للنص مع مراعاة اللغة"""
        normalized = normalize_for_comparison(text)
        return hashlib.md5(f"{lang}:{normalized}".encode()).hexdigest()[:16]
    
    def add_correction(self, wrong: str, correct: str, lang: str,
                      context: str = "", user_rating: float = 1.0,
                      auto_apply: bool = False) -> Dict:
        """
        إضافة تصحيح جديد وتحديث الإحصائيات
        
        Args:
            wrong: النص الأصلي (الخطأ)
            correct: النص المصحح
            lang: رمز اللغة (ar/en/de)
            context: نص محيط للسياق (اختياري)
            user_rating: تقييم المستخدم للتصحيح (0.0-1.0)
            auto_apply: هل تم التطبيق تلقائياً أم يدوياً؟
        
        Returns:
            dict: معلومات عن عملية الإضافة
        """
        wrong_norm = normalize_for_comparison(wrong)
        correct_norm = normalize_for_comparison(correct)
        wrong_hash = self._hash_text(wrong, lang)
        
        conn = sqlite3.connect(self.db_path)
        c = conn.cursor()
        
        # التحقق من وجود التصحيح مسبقاً
        c.execute('SELECT id, confidence, usage_count FROM corrections WHERE wrong_hash=? AND language=?',
                 (wrong_hash, lang))
        existing = c.fetchone()
        
        if existing:
            # تحديث الإحصائيات الموجودة
            corr_id, old_conf, old_count = existing
            new_conf = (old_conf * old_count + user_rating) / (old_count + 1)
            new_count = old_count + 1
            
            c.execute('''
                UPDATE corrections 
                SET correct_text=?, confidence=?, usage_count=?, 
                    user_feedback_score=?, updated_at=CURRENT_TIMESTAMP
                WHERE id=?
            ''', (correct_norm, new_conf, new_count, user_rating, corr_id))
            
            action = 'updated'
        else:
            # إضافة سجل جديد
            initial_conf = 0.9 if auto_apply else 0.7
            c.execute('''
                INSERT INTO corrections 
                (wrong_hash, wrong_text, correct_text, language, context_preview, 
                 confidence, usage_count, user_feedback_score)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?)
            ''', (wrong_hash, wrong_norm, correct_norm, lang, context[:100] if context else "",
                  initial_conf, 1, user_rating))
            corr_id = c.lastrowid
            action = 'created'
        
        # تسجيل في التاريخ للتراجع
        c.execute('INSERT INTO correction_history (correction_id, action) VALUES (?, ?)',
                 (corr_id, action))
        
        conn.commit()
        conn.close()
        
        # تحديث الذاكرة المؤقتة
        cache_key = f"{lang}:{wrong_hash}"
        self._memory_cache[cache_key] = {
            "correct": correct_norm,
            "confidence": new_conf if existing else initial_conf,
            "count": new_count if existing else 1
        }
        
        # مزامنة اختيارية
        if self.sync_enabled:
            self._queue_sync()
        
        return {
            "status": "success",
            "action": action,
            "correction_id": corr_id,
            "confidence": new_conf if existing else initial_conf,
            "can_auto_apply": (new_conf if existing else initial_conf) >= self.MIN_CONFIDENCE_FOR_AUTO
        }
    
    def suggest_correction(self, text: str, lang: str, 
                          top_k: int = 5) -> List[Dict]:
        """
        اقتراح تصحيحات محتملة للنص المدخل
        
        Returns:
            list: قائمة بالاقتراحات مرتبة حسب الثقة
        """
        text_norm = normalize_for_comparison(text)
        wrong_hash = self._hash_text(text, lang)
        suggestions = []
        
        # 1. بحث مباشر بالبصمة
        cache_key = f"{lang}:{wrong_hash}"
        if cache_key in self._memory_cache:
            cached = self._memory_cache[cache_key]
            suggestions.append({
                "text": cached["correct"],
                "confidence": cached["confidence"],
                "match_type": "exact_hash",
                "usage_count": cached["count"]
            })
        
        # 2. بحث في قاعدة البيانات للكلمات المشابهة
        conn = sqlite3.connect(self.db_path)
        c = conn.cursor()
        
        # جلب جميع التصحيحات لنفس اللغة
        c.execute('SELECT wrong_text, correct_text, confidence, usage_count FROM corrections WHERE language=?', (lang,))
        for row in c.fetchall():
            stored_wrong, stored_correct, conf, count = row
            
            # حساب التشابه للنصوص العربية/اللاتينية
            sim = arabic_normalized_similarity(text_norm, normalize_for_comparison(stored_wrong))
            
            if sim > 0.65 and stored_correct not in [s["text"] for s in suggestions]:
                # دمج التشابه مع ثقة التصحيح المخزن
                combined_conf = sim * 0.4 + conf * 0.6
                
                if combined_conf > 0.5:  # عتبة الحد الأدنى
                    suggestions.append({
                        "text": stored_correct,
                        "confidence": combined_conf,
                        "match_type": "fuzzy",
                        "similarity": sim,
                        "usage_count": count
                    })
        
        conn.close()
        
        # 3. ترتيب وإرجاع الأفضل
        suggestions.sort(key=lambda x: x["confidence"], reverse=True)
        return suggestions[:top_k]
    
    def can_auto_correct(self, text: str, lang: str) -> Tuple[bool, Optional[str], float]:
        """
        التحقق مما إذا كان يمكن تطبيق تصحيح تلقائي
        
        Returns:
            (can_apply: bool, corrected_text: str|None, confidence: float)
        """
        suggestions = self.suggest_correction(text, lang, top_k=1)
        
        if suggestions and suggestions[0]["confidence"] >= self.MIN_CONFIDENCE_FOR_AUTO:
            return True, suggestions[0]["text"], suggestions[0]["confidence"]
        
        return False, None, suggestions[0]["confidence"] if suggestions else 0.0
    
    def undo_last_correction(self, wrong: str, lang: str) -> bool:
        """التراجع عن آخر تصحيح لكلمة معينة"""
        wrong_hash = self._hash_text(wrong, lang)
        
        conn = sqlite3.connect(self.db_path)
        c = conn.cursor()
        
        # وضع علامة "ملغي" بدلاً من الحذف الفعلي (لأغراض التدقيق)
        c.execute('''
            UPDATE corrections 
            SET confidence = confidence * 0.1, usage_count = usage_count - 1
            WHERE wrong_hash=? AND language=? AND usage_count > 0
        ''', (wrong_hash, lang))
        
        # تسجيل حدث التراجع
        c.execute('''
            INSERT INTO correction_history (correction_id, action)
            SELECT id, 'undone' FROM corrections WHERE wrong_hash=? AND language=?
        ''', (wrong_hash, lang))
        
        affected = c.rowcount
        conn.commit()
        conn.close()
        
        # إزالة من الذاكرة المؤقتة
        cache_key = f"{lang}:{wrong_hash}"
        self._memory_cache.pop(cache_key, None)
        
        return affected > 0
    
    def review_corrections(self, lang: Optional[str] = None, 
                          min_usage: int = 1,
                          sort_by: str = "confidence") -> List[Dict]:
        """
        مراجعة التصحيحات المحفوظة للتحقق منها
        
        Args:
            lang: تصفية حسب اللغة (None = الكل)
            min_usage: الحد الأدنى لعدد مرات الاستخدام
            sort_by: معيار الترتيب (confidence/usage_count/updated_at)
        """
        conn = sqlite3.connect(self.db_path)
        conn.row_factory = sqlite3.Row
        c = conn.cursor()
        
        query = '''
            SELECT wrong_text, correct_text, language, confidence, 
                   usage_count, context_preview, created_at, updated_at
            FROM corrections WHERE usage_count >= ?
        '''
        params = [min_usage]
        
        if lang and lang != "all":
            query += " AND language = ?"
            params.append(lang)
        
        order_map = {
            "confidence": "confidence DESC",
            "usage_count": "usage_count DESC", 
            "updated_at": "updated_at DESC",
            "created_at": "created_at DESC"
        }
        query += f" ORDER BY {order_map.get(sort_by, 'confidence DESC')}"
        
        c.execute(query, params)
        results = [dict(row) for row in c.fetchall()]
        conn.close()
        
        return results
    
    def export_corrections(self, output_path: str, lang_filter: Optional[str] = None) -> str:
        """تصدير التصحيحات بصيغة JSON للمشاركة"""
        corrections = self.review_corrections(lang=lang_filter)
        
        export_data = {
            "version": "1.0",
            "exported_at": time.strftime("%Y-%m-%d %H:%M:%S"),
            "language": lang_filter or "all",
            "count": len(corrections),
            "corrections": [
                {
                    "wrong": c["wrong_text"],
                    "correct": c["correct_text"],
                    "lang": c["language"],
                    "confidence": round(c["confidence"], 3),
                    "usage_count": c["usage_count"]
                }
                for c in corrections
            ]
        }
        
        with open(output_path, 'w', encoding='utf-8') as f:
            json.dump(export_data, f, ensure_ascii=False, indent=2)
        
        return output_path
    
    def import_corrections(self, input_path: str, merge_strategy: str = "trust_higher") -> Dict:
        """
        استيراد تصحيحات من ملف JSON
        
        merge_strategy: 
            - "trust_higher": الاحتفاظ بالتصحيح ذي الثقة الأعلى
            - "always_update": استبدال بالسجل الجديد دائماً
            - "keep_both": حفظ كلا السجلين مع علامة المصدر
        """
        with open(input_path, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        imported = 0
        skipped = 0
        
        for entry in data.get("corrections", []):
            wrong = entry["wrong"]
            correct = entry["correct"]
            lang = entry["lang"]
            new_conf = entry.get("confidence", 0.7)
            
            wrong_hash = self._hash_text(wrong, lang)
            
            conn = sqlite3.connect(self.db_path)
            c = conn.cursor()
            
            c.execute('SELECT confidence FROM corrections WHERE wrong_hash=? AND language=?',
                     (wrong_hash, lang))
            existing = c.fetchone()
            
            should_update = False
            if not existing:
                should_update = True
            elif merge_strategy == "trust_higher" and new_conf > existing[0]:
                should_update = True
            elif merge_strategy == "always_update":
                should_update = True
            
            if should_update:
                if existing:
                    c.execute('''
                        UPDATE corrections SET correct_text=?, confidence=?, updated_at=CURRENT_TIMESTAMP
                        WHERE wrong_hash=? AND language=?
                    ''', (correct, new_conf, wrong_hash, lang))
                else:
                    c.execute('''
                        INSERT INTO corrections (wrong_hash, wrong_text, correct_text, language, confidence)
                        VALUES (?, ?, ?, ?, ?)
                    ''', (wrong_hash, wrong, correct, lang, new_conf))
                imported += 1
            else:
                skipped += 1
            
            conn.commit()
            conn.close()
        
        return {"imported": imported, "skipped": skipped, "total": len(data.get("corrections", []))}
    
    def _queue_sync(self):
        """إضافة مهمة مزامنة إلى الطابور (للتطبيق المستقبلي مع GitHub/Drive)"""
```

## 📋 الخلية الجديدة: "✏️ تصحيح تفاعلي متقدم"

```python
# خلية 18.5 — واجهة التصحيح التفاعلي (كلمة بكلمة / حرف بحرف)
# ===========================================================

import gradio as gr
import json, re, sqlite3
from pathlib import Path
from modules.nlp.correction_learner import CorrectionLearner
from modules.nlp.arabic_nlp_utils import normalize_for_comparison, arabic_normalized_similarity

# تهيئة نظام التعلم (مرة واحدة عند تحميل الواجهة)
learner = CorrectionLearner(sync_enabled=False)  # يمكن تفعيل المزامنة لاحقاً

def tokenize_text(text: str, lang: str = "ar") -> list:
    """
    تقسيم النص إلى وحدات قابلة للتحديد (كلمات/أحرف)
    يدعم: العربية، الإنجليزية، الألمانية
    """
    tokens = []
    
    if lang == "ar":
        # للعربية: تقسيم مع الحفاظ على التشكيل والربط
        pattern = r'([\u0600-\u06FF\u0750-\u077F\u08A0-\u08FF]+|\s+|[^\u0600-\u06FF\s])'
    else:
        # للإنجليزية/الألمانية: تقسيم قياسي
        pattern = r'(\w+|\s+|[^\w\s])'
    
    parts = re.findall(pattern, text)
    
    for i, part in enumerate(parts):
        if part.strip():  # تجاهل المسافات الفارغة في العرض
            tokens.append({
                "text": part,
                "type": "word" if len(part) > 1 or part in ["-", "'"] else "char",
                "index": i,
                "lang": lang,
                "needs_review": False  # سيتم تحديثه لاحقاً
            })
    
    return tokens

def highlight_with_suggestions(text: str, lang: str) -> tuple:
    """
    عرض النص مع تلوين الكلمات التي تحتاج مراجعة + اقتراحات
    """
    tokens = tokenize_text(text, lang)
    highlighted = []
    
    for token in tokens:
        word = token["text"]
        can_auto, suggestion, conf = learner.can_auto_correct(word, lang)
        
        if can_auto and conf >= 0.9:
            # أخضر: تصحيح موثوق تلقائياً
            highlighted.append((word, "correct"))
            token["auto_corrected"] = True
        elif conf > 0.5:
            # أصفر: يحتاج مراجعة
            highlighted.append((word, "uncertain"))
            token["needs_review"] = True
            token["suggestion"] = suggestion
        else:
            # رمادي: لا يوجد اقتراح
            highlighted.append((word, None))
    
    return highlighted, tokens

def on_token_select(token_data, current_output: str, lang: str):
    """
    عند تحديد كلمة/حرف: عرض التفاصيل والاقتراحات
    """
    if not token_data or not isinstance(token_data, dict):
        return "", {}, "اختر عنصراً من النص", current_output, []
    
    word = token_data.get("text", "")
    suggestion = token_data.get("suggestion", "")
    needs_review = token_data.get("needs_review", False)
    
    # جلب اقتراحات إضافية من قاعدة البيانات
    extra_suggestions = learner.suggest_correction(word, lang, top_k=3)
    
    # تحضير قائمة الاقتراحات للعرض
    suggestions_list = [
        {"text": s["text"], "confidence": f"{s['confidence']:.0%}", "source": s.get("match_type", "unknown")}
        for s in extra_suggestions
    ]
    
    # رسالة حالة
    if needs_review and suggestion:
        status = f"💡 اقتراح: `{suggestion}` (ثقة: {extra_suggestions[0]['confidence']:.0%} إذا وجدت)"
    elif not word.strip():
        status = "⚠️ عنصر فارغ"
    else:
        status = f"✅ `{word}` — لا توجد تصحيحات مقترحة"
    
    return word, {"suggestions": suggestions_list}, status, current_output, extra_suggestions

def apply_correction(selected_word: str, correction: str, lang: str, 
                    current_output: str, tokens: list, history: list):
    """
    تطبيق التصحيح: استبدال في النص + حفظ في القاموس + تحديث السجل
    """
    if not selected_word or not correction:
        return current_output, "⚠️ أدخل نص التصحيح أولاً", history, tokens
    
    # حفظ التصحيح للتعلم المستقبلي
    result = learner.add_correction(selected_word, correction, lang, auto_apply=False)
    
    # استبدال الكلمة في النص النهائي (أول ظهور فقط لتجنب الاستبدال العشوائي)
    if current_output and selected_word in current_output:
        new_output = current_output.replace(selected_word, correction, 1)
    else:
        new_output = correction
    
    # تحديث سجل التراجع
    new_history = history + [{"action": "correct", "from": selected_word, "to": correction, "output": current_output}]
    
    # تحديث حالة الكلمة في القائمة
    for t in tokens:
        if t["text"] == selected_word:
            t["auto_corrected"] = True
            t["needs_review"] = False
            break
    
    status = f"✅ حُفظ: `{selected_word}` → `{correction}` (ثقة: {result['confidence']:.0%})"
    return new_output, status, new_history, tokens

def reject_word(selected_word: str, lang: str, current_output: str, 
               tokens: list, history: list):
    """
    رفض/حذف الكلمة: إزالتها من السياق + تسجيل كـ "غير مرغوبة"
    """
    # تسجيل الكلمة كـ "blacklisted" إذا كانت عديمة المعنى
    learner.add_correction(selected_word, "", lang, user_rating=0.0)
    
    # إزالة الكلمة من النص
    if current_output and selected_word in current_output:
        new_output = current_output.replace(selected_word, "", 1).strip()
    else:
        new_output = current_output
    
    # تحديث السجل
    new_history = history + [{"action": "delete", "word": selected_word, "output": current_output}]
    
    # تحديث حالة الكلمة
    for t in tokens:
        if t["text"] == selected_word:
            t["deleted"] = True
            break
    
    return new_output, f"🗑️ تم حذف `{selected_word}`", new_history, tokens

def undo_last(history: list, current_output: str):
    """
    التراجع عن آخر إجراء
    """
    if not history:
        return current_output, "⚠️ لا يوجد إجراءات للتراجع", history
    
    last = history[-1]
    previous_output = last.get("output", current_output)
    
    # إعادة الثقة للتصحيح إذا كان تصحيحاً
    if last.get("action") == "correct":
        learner.undo_last_correction(last["from"], "ar")  # يمكن تحسينها لاكتشاف اللغة
    
    return previous_output, f"↩️ تم التراجع عن: {last.get('action', 'unknown')}", history[:-1]

def load_correction_review(lang_filter: str = "all", min_usage: int = 1):
    """

        datatype=["str", "str", "str", "number", "str"],
        interactive=False,
        row_count=10
    )
```

> 🤔 **سؤال توجيهي**: هل تود أن أضيف ميزة **التصحيح على مستوى الأحرف** (مثلاً: تحديد حرف "ب" وتصحيحه إلى "ت" داخل كلمة)؟ هذا يتطلب معالجة خاصة للنصوص العربية المتصلة.

أو ننتقل مباشرة لاختبار الواجهة؟ 🚀

### USER
نعم مستوى الأحرف ثم اختبار الواجهة 

### ASSISTANT
ممتاز. قبل الغوص في الكود، دعني أطرح سؤالاً توجيهياً:

> 🤔 **سؤال سقراطي**: النص العربي متصل وتتشكل الأحرف حسب موقعها (بداية، وسط، نهاية، منعزلة). عندما يصحح المستخدم حرفاً واحداً مثل `ب` → `ت`، هل يكفي حفظ هذا الزوج، أم يجب حفظ **السياق المحيط** (مثلاً: `ـبـ` → `ـتـ`) لتجنب كسر الربط البصري؟ وكيف سيتعامل النظام مع التشكيل (الفتحة، الكسرة) إذا رُبط بحرف أم عُدّ ككيان مستقل؟

سأصمم الكود بحيث يتعامل مع **المجموعات الشكلية (Grapheme Clusters)** كوحدة أساسية، مما يحافظ على ربط الأحرف مع التشكيل تلقائياً، ويخزن التصحيحات مع الوزن السياقي.

```python
# خلية 18.5 — ✏️ تصحيح حرف بحرف (Character-Level Interactive Correction)
# ======================================================================

import gradio as gr
import re, json, sqlite3, time
from modules.nlp.correction_learner import CorrectionLearner
from modules.nlp.arabic_nlp_utils import normalize_for_comparison

# تهيئة المتعلم (يستخدم نفس قاعدة البيانات السابقة)
learner = CorrectionLearner(db_path="artifacts/corrections.db", sync_enabled=False)

def tokenize_graphemes(text: str) -> list:
    """تقسيم النص إلى وحدات حرفية+تشكيل مع الحفاظ على المؤشرات الأصلية"""
    # يطابق: حرف أساسي + تشكيله + علامات خاصة، أو مسافات، أو أي رمز
    pattern = r'[\u0600-\u06FF\u0750-\u077F\u08A0-\u08FF]+(?:[\u0610-\u061A\u064B-\u065F\u0670]*)?|\s+|.'
    tokens = []
    for m in re.finditer(pattern, text):
        tokens.append({
            "text": m.group(),
            "start": m.start(),
            "end": m.end(),
            "is_space": m.group().isspace()
        })
    return tokens

def prepare_highlight_ui(text: str, lang: str = "ar") -> tuple:
    """تجهيز النص للعرض مع تلوين الأحرف التي تحتاج مراجعة"""
    chars = tokenize_graphemes(text)
    highlighted = []
    review_indices = []
    
    for i, c in enumerate(chars):
        if c["is_space"]:
            highlighted.append((c["text"], None))
            continue
            
        # فحص سريع: هل الحرف/المجموعة سبق تصحيحها؟
        can_auto, suggested, conf = learner.can_auto_correct(c["text"], lang)
        if can_auto and conf >= 0.9:
            highlighted.append((c["text"], "correct"))
        elif conf > 0.4:
            highlighted.append((c["text"], "uncertain"))
            review_indices.append(i)
        else:
            highlighted.append((c["text"], None))
            
    return highlighted, chars, review_indices

def on_char_select(select_ gr.SelectData, current_output: str, history: list):
    """عند النقر على حرف: عرضه واقتراح بدائل"""
    if not select_data or not select_data.value:
        return "", {}, "اختر حرفاً من النص", current_output, history, []
        
    selected_text, label = select_data.value
    idx = select_data.index
    
    # جلب اقتراحات من قاعدة التصحيحات
    suggestions = learner.suggest_correction(selected_text, "ar", top_k=5)
    alt_list = [{"char": s["text"], "confidence": f"{s['confidence']:.0%}", "source": s.get("match_type", "db")} for s in suggestions]
    
    status = f"📍 مؤشر: {idx} | النص: `{selected_text}`"
    if suggestions:
        status += f" | 💡 أقرب اقتراح: `{suggestions[0]['text']}`"
        
    return selected_text, {"alternatives": alt_list}, status, current_output, history, suggestions

def apply_char_correction(original: str, replacement: str, lang: str, 
                          current_output: str, history: list, suggestions: list):
    """تطبيق التصحيح الحرفي + حفظ للتعلم + تحديث السجل"""
    if not original or not replacement:
        return current_output, "⚠️ أدخل البديل", history, []
        
    # حفظ في قاعدة التعلم
    result = learner.add_correction(original, replacement, lang, auto_apply=False)
    
    # استبدال أول ظهور في النص النهائي
    if original in current_output:
        new_output = current_output.replace(original, replacement, 1)
    else:
        new_output = replacement
        
    history.append({"type": "char_fix", "from": original, "to": replacement, "prev_output": current_output})
    return new_output, f"✅ `{original}` → `{replacement}` (ثقة: {result['confidence']:.0%})", history, []

def delete_char(original: str, current_output: str, history: list):
    """حذف حرف عديم المعنى"""
    if not original: return current_output, "⚠️ لا يوجد حرف محدد", history
    
    learner.add_correction(original, "", "ar", user_rating=0.0)
    if original in current_output:
        new_out = current_output.replace(original, "", 1).strip()
    else:
        new_out = current_output
        
    history.append({"type": "delete", "char": original, "prev_output": current_output})
    return new_out, f"🗑️ تم حذف `{original}`", history

def undo_last(history: list, current_output: str):
    """التراجع عن آخر إجراء حرفي"""
    if not history:
        return current_output, "⚠️ لا يوجد إجراءات للتراجع", history
        
    last = history.pop()
    prev = last.get("prev_output", current_output)
    return prev, f"↩️ تراجع عن: {last.get('type')}", history

def copy_predicted_to_corrected(predicted: str, current_corrected: str) -> str:
    """زر نسخ سريع: نسخ النص المتوقع كاملاً إلى خانة المصحح إذا كان صحيحاً"""
    return predicted if predicted.strip() else current_corrected

with gr.Blocks() as char_correction_ui:

    char_history = gr.State([])

        with gr.Column(scale=2):
            predicted_input = gr.Textbox(label="📥 النص المتوقع من OCR", lines=4, rtl=True, interactive=True)

    status_char = gr.Textbox(label="🔔 الحالة", interactive=False)
```

## 🧠 الانتقال لتحسين خوارزمية الاقتراحات

الآن وقد أصبح النظام قادراً على **التعلم، التصحيح، والمزامنة**، السؤال الجوهري للخوارزمية هو:

> 🤔 **سؤال سقراطي للخوارزمية**: عندما يصحح المستخدم `ك → ق` في كلمة `كتابة`، هل يجب أن نعتبرها قاعدة حرفية عامة، أم أن نربطها بـ `السياق المجاور` (مثلاً: إذا جاء بعد `ك` حرف `ت`، فالأرجح أن المقصود `ق`)؟ وكيف نقيس `وزن الثقة` ديناميكياً بناءً على تكرار التصحيح عبر مستخدمين مختلفين؟

سأقترح تحسين الخوارزمية باستخدام **نموذج ن-غرام (N-Gram) خفيف + تشابه ليفنشتاين المُعدّل للعربية**، مع نظام أوزان يتكيف مع التكرار والسياق. هل تريد أن أبني خوارزمية الاقتراحات الذكية في الخلية التالية (`18.9`) مع دعم:
- ✅ التشابه الصوتي/الشكلي للعربية
- ✅ الوزن السياقي (الحروف المجاورة)
- ✅ التكيف التلقائي مع التكرار
- ✅ عرض سبب الاقتراح في الواجهة؟

أخبرني وسأنتقل للتنفيذ مباشرة. 🚀

### USER
نعم 

### ASSISTANT
ممتاز. سننتقل الآن من **الفكرة** إلى **التنفيذ الفعلي الآمن والقابل للتوسع**.

> 🤔 **سؤال سقراطي تمهيدي**: المزامنة التلقائية تعني نقل بيانات حساسة (قاموس تصحيحات) عبر الشبكة. هل نفضل `المزامنة عند الحفظ (Sync-on-Save)` لتقليل الحمل الشبكي، أم `المزامنة الدورية (Background Sync)` مع طابور انتظار؟ وكيف نتعامل مع تضارب النسخ إذا عدّلت نفس الكلمة من جهازين مختلفين في نفس اللحظة؟

سأفترض الإجابة العملية الآمنة: **مزامنة يدوية/شبه-تلقائية عند الطلب + دمج يعتمد على `updated_at` مع سجل تضارب**، وسأبني الكود accordingly.

## 🧠 خلية 18.9 — 📊 خوارزمية الاقتراحات الذكية (Context-Aware)

```python
# خلية 18.9 — 💡 SmartSuggestionEngine: اقتراحات تعتمد على السياق والتردد
# ======================================================================

import re
from collections import defaultdict
from modules.nlp.arabic_nlp_utils import normalize_for_comparison, arabic_normalized_similarity

class SmartSuggestionEngine:
    def __init__(self, db_path="artifacts/corrections.db"):
        self.db_path = Path(db_path)
        self._cache = defaultdict(list)
        
    def _load_corrections(self):
        if not self.db_path.exists(): return []
        conn = sqlite3.connect(self.db_path)
        conn.row_factory = sqlite3.Row
        data = conn.execute("SELECT * FROM corrections WHERE usage_count > 0").fetchall()
        conn.close()
        return [dict(r) for r in data]
        
    def get_context_window(self, text, lang, window=2):
        """استخراج نافذة السياق المحيط (كلمات/أحرف قبل وبعد)"""
        tokens = re.findall(r'\S+', text) if lang != 'ar' else list(text.replace(' ', ''))
        idx = tokens.index(text) if text in tokens else -1
        if idx < 0: return "", ""
        before = " ".join(tokens[max(0, idx-window):idx])
        after  = " ".join(tokens[idx+1:idx+1+window])
        return before.strip(), after.strip()
        
    def score_suggestion(self, target, candidate_row, lang):
        """حساب درجة الاقتراح الموزونة"""
        wrong = candidate_row['wrong_text']
        sim = arabic_normalized_similarity(target, normalize_for_comparison(wrong))
        freq_weight = min(candidate_row['usage_count'] / 5.0, 1.0)  # سقف عند 5 استخدامات
        user_rating = candidate_row.get('user_feedback_score', 0.8)
        
        # الوزن النهائي
        score = (sim * 0.45) + (freq_weight * 0.30) + (user_rating * 0.25)
        return round(score, 3)
        
    def suggest(self, target_text, lang, context_text="", top_k=5):
        """إرجاع قائمة اقتراحات مفسرة"""
        corrections = self._load_corrections()
        suggestions = []
        
        # تطبيع الهدف للمقارنة
        target_norm = normalize_for_comparison(target_text)
        
        for row in corrections:
            if row['language'] != lang: continue
            
            score = self.score_suggestion(target_text, row, lang)
            if score < 0.5: continue  # عتبة الاستبعاد
            
            # استخراج سبب الاقتراح
            reason = "تشابه عالي" if score > 0.85 else "تكرار استخدام" if row['usage_count'] > 2 else "تقييم مستخدم"
            
            suggestions.append({
                "text": row['correct_text'],
                "confidence": score,
                "reason": reason,
                "source": row.get('context_preview', 'عام')[:20]
            })
            
        # ترتيب وإرجاع الأفضل
        suggestions.sort(key=lambda x: x['confidence'], reverse=True)
        return suggestions[:top_k]

# تهيئة المحرك
suggest_engine = SmartSuggestionEngine()
print("✅ SmartSuggestionEngine جاهز.")
```

### ASSISTANT
ممتاز. هذا النهج الهجين **أذكى عملياً**: يبدأ بخفة وسرعة، ويمنحك عمقاً تحليلياً عند الحاجة، دون استنزاف موارد كولاب المجاني.

سأبني لك `HybridSuggestionEngine` يعتمد على:
- **الوضع السريع (Fast Mode)**: مطابقة شكلية موزونة (Levenshtein/N-gram) + تردد التصحيح + ثقة المستخدم. `⏱️ <10ms`
- **الوضع العميق (Deep Mode)**: تحليل السياق المجاور (Bigram/Trigram) + التشابه الصوتي/الجذري للعربية + مصفوفة التواجد المشترك. `⏱️ ~50-100ms`

كلا الوضعين يعملان **بدون نماذج ثقيلة**، ويعيدان تفصيلاً شفافاً لوزن كل عامل لاتخاذ القرار بوضوح.

```python
# خلية 18.9.1 — 🧠 HybridSuggestionEngine: اقتراحات سريعة + تحليل عميق عند الطلب
# ==============================================================================

import re, sqlite3, json
from collections import Counter
from modules.nlp.arabic_nlp_utils import normalize_for_comparison, arabic_normalized_similarity

class HybridSuggestionEngine:
    def __init__(self, db_path="artifacts/corrections.db"):
        self.db_path = db_path
        self._context_cache = {}

    def _fetch_corrections(self):
        """جلب التصحيحات من SQLite مرة واحدة وتحسينها للبحث"""
        if not self.db_path: return []
        conn = sqlite3.connect(self.db_path)
        conn.row_factory = sqlite3.Row
        rows = conn.execute("SELECT * FROM corrections WHERE usage_count > 0").fetchall()
        conn.close()
        return [dict(r) for r in rows]

    def _extract_context(self, full_text, target, window=3):
        """استخراج نافذة السياق (كلمات قبل/بعد)"""
        tokens = re.findall(r'\S+', full_text or "")
        idx = next((i for i, t in enumerate(tokens) if target in t), -1)
        if idx < 0: return [], []
        start, end = max(0, idx-window), min(len(tokens), idx+window+1)
        return tokens[start:idx], tokens[idx+1:end]

    def _score_fast(self, target_norm, cand, lang):
        """الوضع السريع: شكل + تردد + ثقة"""
        wrong_norm = normalize_for_comparison(cand['wrong_text'])
        sim = arabic_normalized_similarity(target_norm, wrong_norm)
        freq = min(cand['usage_count'] / 5.0, 1.0)
        conf = cand.get('confidence', 0.5)
        
        score = (sim * 0.50) + (freq * 0.30) + (conf * 0.20)
        return round(score, 3), {
            "string_match": round(sim, 2),
            "frequency_weight": round(freq, 2),
            "user_confidence": round(conf, 2),
            "mode": "FAST"
        }

    def _score_deep(self, target_norm, cand, lang, context_before, context_after):
        """الوضع العميق: سياق + جذور/تشابه صوتي + تواجد مشترك"""
        sim_base = arabic_normalized_similarity(target_norm, normalize_for_comparison(cand['wrong_text']))
        
        # 1. فحص التواجد المشترك في السياق (Co-occurrence)
        wrong_ctx = cand.get('context_preview', "")
        ctx_match = 0.0
        if context_before and wrong_ctx:
            ctx_match = sum(1 for w in context_before if w in wrong_ctx) / len(context_before)
        if context_after and wrong_ctx:
            ctx_match += sum(1 for w in context_after if w in wrong_ctx) / len(context_after)
        ctx_match = min(ctx_match, 1.0) * 0.15  # وزن 15%

        # 2. وزن إضافي للتشكيل/الأحرف المتصلة (محاكاة جذرية خفيفة)
        morph_weight = 0.05 if any(c in target_norm for c in "بتثجحخدذرزسشصضطظعغفقكلمنهوي") else 0.0

        # 3. حساب الدرجة النهائية
        score = (sim_base * 0.40) + (min(cand['usage_count']/4.0, 1.0) * 0.30) + \
                (cand.get('confidence', 0.5) * 0.15) + ctx_match + morph_weight
                
        return round(score, 3), {
            "string_match": round(sim_base, 2),
            "context_cooccurrence": round(ctx_match, 2),
            "frequency_weight": round(min(cand['usage_count']/4.0, 1.0), 2),
            "user_confidence": round(cand.get('confidence', 0.5), 2),
            "mode": "DEEP"
        }

    def suggest(self, target_text, lang, full_context="", mode="fast", top_k=5):
        """إرجاع اقتراحات مع تفصيل الأوزان"""
        corrections = self._fetch_corrections()
        target_norm = normalize_for_comparison(target_text)
        ctx_b, ctx_a = self._extract_context(full_context, target_text)
        results = []

        for cand in corrections:
            if cand['language'] != lang: continue
            if mode == "deep":
                score, breakdown = self._score_deep(target_norm, cand, lang, ctx_b, ctx_a)
            else:
                score, breakdown = self._score_fast(target_norm, cand, lang)
                
            if score < 0.45: continue  # عتبة الاستبعاد
            
            results.append({
                "text": cand['correct_text'],
                "score": score,
                "breakdown": breakdown,
                "source_preview": cand.get('context_preview', 'N/A')[:25]
            })

        results.sort(key=lambda x: x['score'], reverse=True)
        return results[:top_k]

# تهيئة المحرك
hybrid_engine = HybridSuggestionEngine()
print("✅ HybridSuggestionEngine جاهز. الوضع الافتراضي: FAST | متاح: DEEP")
```

## 📦 خلية 18.9.1 — `UnifiedSuggestionEngine` (أعلى درجة + تحليل شفاف)

```python
# خلية 18.9.1 — 🎯 UnifiedSuggestionEngine: أعلى درجة دائماً + تفاصيل قابلة للتوسيع
# ==============================================================================

import re, sqlite3
from modules.nlp.arabic_nlp_utils import normalize_for_comparison, arabic_normalized_similarity

class UnifiedSuggestionEngine:
    def __init__(self, db_path="artifacts/corrections.db"):
        self.db_path = db_path

    def _fetch_corrections(self):
        """جلب التصحيحات الفعالة من قاعدة البيانات"""
        if not self.db_path: return []
        conn = sqlite3.connect(self.db_path)
        conn.row_factory = sqlite3.Row
        rows = conn.execute("SELECT * FROM corrections WHERE usage_count > 0").fetchall()
        conn.close()
        return [dict(r) for r in rows]

    def _extract_context(self, full_text, target, window=3):
        """استخراج نافذة السياق (كلمات قبل وبعد الهدف)"""
        tokens = re.findall(r'\S+', full_text or "")
        idx = next((i for i, t in enumerate(tokens) if target in t), -1)
        if idx < 0: return [], []
        return tokens[max(0, idx-window):idx], tokens[idx+1:idx+1+window]

    def get_top_suggestion(self, target_text, lang, full_context=""):
        """إرجاع الاقتراح الأعلى درجة مع تفصيل الأوزان"""
        corrections = self._fetch_corrections()
        target_norm = normalize_for_comparison(target_text)
        ctx_b, ctx_a = self._extract_context(full_context, target_text)
        results = []

        for cand in corrections:
            if cand['language'] != lang: continue
            wrong_norm = normalize_for_comparison(cand['wrong_text'])

            # 1️⃣ التطابق الشكلي (تطبيع + ليفنشتاين خفيف)
            sim = arabic_normalized_similarity(target_norm, wrong_norm)
            
            # 2️⃣ التردد (مقاس 0-1، سقف عند 5 استخدامات)
            freq = min(cand['usage_count'] / 5.0, 1.0)
            
            # 3️⃣ ثقة المستخدم التاريخية
            conf = cand.get('confidence', 0.5)
            
            # 4️⃣ توافق السياق المجاور (يُضاف فقط إذا وُجد سياق مدخل)
            ctx_score = 0.0
            if (ctx_b or ctx_a) and cand.get('context_preview'):
                preview = cand['context_preview']
                b_match = sum(1 for w in ctx_b if w in preview) / max(len(ctx_b), 1)
                a_match = sum(1 for w in ctx_a if w in preview) / max(len(ctx_a), 1)
                ctx_score = ((b_match + a_match) / 2)
            
            # 🧮 الدرجة النهائية الموحدة (مجموع الأوزان = 1.0)
            final_score = (sim * 0.35) + (freq * 0.25) + (conf * 0.20) + (ctx_score * 0.20)
            
            if final_score < 0.45: continue  # عتبة الاستبعاد
            
            results.append({
                "text": cand['correct_text'],
                "score": round(final_score, 3),
                "breakdown": {
                    "التطابق الشكلي": round(sim, 2),
                    "تكرار التصحيح (التردد)": round(freq, 2),
                    "تقييم المستخدم السابق (الثقة)": round(conf, 2),
                    "توافق السياق المجاور": round(ctx_score, 2)
                },
                "reason": "✅ سياق قوي" if ctx_score > 0.4 else "📈 شكل وتردد عالي"
            })

        # ترتيب وإرجاع الأعلى فقط
        results.sort(key=lambda x: x['score'], reverse=True)
        return results[0] if results else None

# تهيئة المحرك الموحد
unified_engine = UnifiedSuggestionEngine()
print("✅ UnifiedSuggestionEngine جاهز. يعرض دائماً الأعلى درجة مع تحليل شفاف.")
```

```python
import gradio as gr, re, json, time
from modules.nlp.arabic_nlp_utils import normalize_for_comparison

# تأكد من أن المحركات مُهيأة مسبقاً (من الخلايا 18.8 و 18.9.1)

def post_process_text(raw_text: str, lang: str = "ar"):
    """معالجة ما بعد الـ OCR: تطبيق تلقائي (>0.85) + تحضير للمراجعة (<0.85)"""
    words = re.findall(r'\S+', raw_text)
    final_tokens = []
    review_items = []

        if not w.strip(): continue
        suggestion = unified_engine.get_top_suggestion(w, lang, full_context=raw_text)
        
        if suggestion and suggestion['score'] >= 0.85:
            # ✅ تطبيق تلقائي صامت

            review_items.append({

                "score": suggestion['score'],
                "breakdown": suggestion['breakdown']
            })
        else:
            # ⚪ لا اقتراح فوق العتبة
            final_tokens.append(w)
            
    return " ".join(final_tokens), review_items

def on_word_select(review_data_json, selected_index):
    """عند النقر على كلمة تحتاج مراجعة: عرض الاقتراح الأعلى + سبب الشفاف"""
```

```python
# خلية 21 — 🗄️ Database Abstraction + Rate Limiter + Smart Cache
# ===========================================================
import os, time, sqlite3, hashlib
from collections import OrderedDict
from functools import wraps
from typing import Optional

class OmniDB:
    """طبقة مجردة لقواعد البيانات: تدعم SQLite للتطوير و PostgreSQL/MySQL للإنتاج"""
    def __init__(self, db_url: Optional[str] = None):
        self.db_url = db_url or os.getenv("OMNIFILE_DB_URL", "sqlite:///artifacts/omnifile.db")
        self._ensure_path()

    def _ensure_path(self):
        if self.db_url.startswith("sqlite"):
            path = self.db_url.replace("sqlite:///", "")
            os.makedirs(os.path.dirname(path), exist_ok=True)

    def get_connection(self):
        if self.db_url.startswith("sqlite"):
            return sqlite3.connect(self.db_url.replace("sqlite:///", ""))
        elif self.db_url.startswith("postgresql"):
            import psycopg2
            return psycopg2.connect(self.db_url.replace("postgresql://", ""))
        elif self.db_url.startswith("mysql"):
            import mysql.connector
            return mysql.connector.connect(**self._parse_mysql())
        raise ValueError("Unsupported DB URL. Use sqlite:///, postgresql://, or mysql://")

    def _parse_mysql(self):
        # تحليل بسيط لـ mysql://user:pass@host/db
        return {"host": "localhost", "user": "root", "password": "", "database": "omnifile"}

db_manager = OmniDB()

class PerformanceOptimizer:
    """كاش LRU + Rate Limiter للتحكم في استهلاك الموارد"""
    def __init__(self, max_cache_size=150, max_rpm=12):
        self.cache = OrderedDict()
        self.max_cache = max_cache_size
        self.rpm_limit = max_rpm
        self.requests = []

    def _cache_key(self, *args, **kwargs):
        raw = str(args) + str(sorted(kwargs.items()))
        return hashlib.md5(raw.encode()).hexdigest()

    def get_cached(self, *args, **kwargs):
        key = self._cache_key(*args, **kwargs)
        if key in self.cache:
            self.cache.move_to_end(key)
            return self.cache[key]
        return None

    def set_cache(self, *args, value, **kwargs):
        key = self._cache_key(*args, **kwargs)
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
        if len(self.cache) > self.max_cache:
            self.cache.popitem(last=False)

    def check_rate_limit(self) -> bool:
        now = time.time()
        self.requests = [t for t in self.requests if now - t < 60]
        if len(self.requests) < self.rpm_limit:
            self.requests.append(now)
            return True
        return False

optimizer = PerformanceOptimizer(max_cache_size=150, max_rpm=12)

def rate_limited_and_cached(func):
    """Decorator لدمج الكاش ومحدد المعدل تلقائياً في دوال OCR/NLP"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        if not optimizer.check_rate_limit():
            raise Exception("⚠️ تجاوزت حد الطلبات المسموح (12/دقيقة). انتظر قليلاً أو حدث التوكن.")
        cached = optimizer.get_cached(*args, **kwargs)
        if cached:
            return cached
        result = func(*args, **kwargs)
        optimizer.set_cache(*args, value=result, **kwargs)
        return result
    return wrapper

print("✅ OmniDB + PerformanceOptimizer جاهزان. \n💡 لتفعيل PostgreSQL: اضبط env var OMNIFILE_DB_URL=postgresql://user:pass@host/db")
```

```python
# 3️⃣ سجل تدقيق غير قابل للتعديل (Audit Chain + تشفير)
# ==========================================
class SecureAuditChain:
    def __init__(self, log_path="artifacts/audit_chain.json"):
        self.log_path = Path(log_path)
        self._init_chain()
        
    def _init_chain(self):
        if not self.log_path.exists():
            self.log_path.parent.mkdir(parents=True, exist_ok=True)
            genesis = {"timestamp": str(datetime.datetime.utcnow()), "action": "GENESIS", "details": "OmniFile v5.0 Audit Init", "prev_hash": "0"*64}
            genesis["current_hash"] = self._hash_entry(genesis)
            with open(self.log_path, "w") as f: json.dump([genesis], f)
            
    def _hash_entry(self, entry):
        raw = json.dumps({k:v for k,v in entry.items() if k!="current_hash"}, sort_keys=True)
        return hashlib.sha256(raw.encode()).hexdigest()
        
    def append(self, action, details, user="system"):
        with open(self.log_path) as f: chain = json.load(f)
        prev_hash = chain[-1]["current_hash"]
        entry = {"timestamp": str(datetime.datetime.utcnow()), "action": action, "details": details, "user": user, "prev_hash": prev_hash}
        entry["current_hash"] = self._hash_entry(entry)
        chain.append(entry)
        with open(self.log_path, "w") as f: json.dump(chain, f, ensure_ascii=False, indent=2)
        return entry["current_hash"]
        
    def verify_integrity(self):
        with open(self.log_path) as f: chain = json.load(f)
        for i in range(1, len(chain)):
            expected = self._hash_entry(chain[i])
            if expected != chain[i]["current_hash"]: return False, f"Tampered at index {i}"
            if chain[i]["prev_hash"] != chain[i-1]["current_hash"]: return False, f"Chain broken at index {i}"
        return True, "✅ السجل سليم وغير مُعدّل"

audit_log = SecureAuditChain()

# ==========================================
# 4️⃣ نسخ احتياطي مُشفّر تلقائي
# ==========================================
class EncryptedBackupManager:
    def __init__(self, backup_dir="artifacts/backups"):
        self.backup_dir = Path(backup_dir)
        self.backup_dir.mkdir(parents=True, exist_ok=True)
        self.key = os.getenv("BACKUP_KEY", Fernet.generate_key().decode())
        self.cipher = Fernet(self.key.encode())
        
    def encrypt_and_save(self, source_path, label="auto"):
        with open(source_path, "rb") as f: data = f.read()
        enc_data = self.cipher.encrypt(data)
        ts = datetime.datetime.utcnow().strftime("%Y%m%d_%H%M")
        out_path = self.backup_dir / f"{label}_{ts}.enc"
        with open(out_path, "wb") as f: f.write(enc_data)
        audit_log.append("BACKUP_CREATED", {"file": str(out_path), "source": str(source_path)})
        return out_path
        
    def restore(self, enc_path, out_path):
        with open(enc_path, "rb") as f: enc_data = f.read()
        data = self.cipher.decrypt(enc_data)
        with open(out_path, "wb") as f: f.write(data)
        audit_log.append("BACKUP_RESTORED", {"file": str(enc_path), "target": str(out_path)})
        return True

backup_mgr = EncryptedBackupManager()

# ==========================================
# 🔗 Decorator موحد للمراقبة والتتبع
# ==========================================
def tracked(operation_name: str = "unknown"):
    def decorator(func):
        def wrapper(*args, **kwargs):
            METRIC_REQUESTS.labels(status="started", lang="ar").inc()
            start = time.time()
            try:
                result = func(*args, **kwargs)
                METRIC_REQUESTS.labels(status="success", lang="ar").inc()
                audit_log.append("OP_SUCCESS", {"operation": operation_name, "input_size": len(str(args))})
                return result
            except Exception as e:
                METRIC_ERRORS.labels(operation=operation_name).inc()
                METRIC_REQUESTS.labels(status="error", lang="ar").inc()
                logger.exception(f"❌ Error in {operation_name}: {e}")
                audit_log.append("OP_ERROR", {"operation": operation_name, "error": str(e)})
                raise
            finally:
                METRIC_LATENCY.observe(time.time() - start)
        return wrapper
    return decorator

print("✅ OmniObservability جاهز:\n  • 📝 Loguru + Sentry (اختياري)\n  • 📊 Prometheus Metrics (:9090/metrics)\n  • 🔒 Audit Chain + تشفير Fernet\n  • 🛡️ @tracked decorator للمراقبة التلقائية")
```

 * **التصويت الجماعي المشفر:** يحول النظام من مجرد أداة معالجة إلى "ذكاء جمعي" يتطور مع الوقت دون الكشف عن هوية المصححين.

```python
# خلية 28 — 📚 Research Integration Layer (Zotero API)
# =========================================================================
import os
from pyzotero import zotero

class ResearchManager:
    def __init__(self, library_id: str = None, library_type: str = 'user', api_key: str = None):
        """
        إدارة المزامنة مع Zotero.
        يمكن جلب الإعدادات من ملف Secrets أو من مدخلات المستخدم المشفرة.
        """
        self.library_id = library_id or os.getenv("ZOTERO_ID")
        self.library_type = library_type
        self.api_key = api_key or os.getenv("ZOTERO_API_KEY")
        
        if self.library_id and self.api_key:
            self.zot = zotero.Zotero(self.library_id, self.library_type, self.api_key)
        else:
            self.zot = None

    def export_to_zotero(self, text_content: str, title: str, item_type: str = 'journalArticle'):
        """تصدير النص المستخرج كعنصر جديد في Zotero"""
        if not self.zot:
            return {"status": "error", "message": "Zotero API credentials missing"}
        
        # إنشاء هيكل العنصر (Template)
        template = self.zot.item_template(item_type)
        template['title'] = title
        template['abstractNote'] = text_content[:1000]  # وضع أول 1000 حرف في الملخص
        
        # إضافة وسم (Tag) لتمييز الملفات المعالجة بواسطة OmniFile
        template['tags'] = [{'tag': 'OmniFile-Processed'}]
        
        resp = self.zot.create_items([template])
        return {"status": "success", "item_id": resp['successful']['0']['key']}

# ------------------------------------------------------------
# دالة الربط مع واجهة Gradio
# ------------------------------------------------------------
def sync_research_task(user_id, text, doc_title, z_id, z_key):
    """دالة وسيطة للتعامل مع طلبات المستخدمين المتعددين"""
    try:
        rm = ResearchManager(library_id=z_id, api_key=z_key)
        result = rm.export_to_zotero(text, doc_title)
        return f"✅ تم التصدير بنجاح! معرف العنصر: {result.get('item_id')}"
    except Exception as e:
        return f"❌ خطأ في المزامنة: {str(e)}"

print("✅ ResearchSync جاهز. يدعم الآن Zotero (قريباً Mendeley).")
```

### 📦 خلية 29 — 📊 AdminCore: مراقبة النظام وتحليلات الاستخدام

```python
# خلية 29 — 🛡️ Admin & System Monitoring Module
# =========================================================================
import psutil
import platform
import time
from redis import Redis
import os

class AdminManager:
    def __init__(self, redis_url: str = None):
        self.redis = Redis.from_url(redis_url) if redis_url else None

    def get_system_stats(self):
        """جلب استهلاك الموارد الفعلي للحاوية"""
        stats = {
            "cpu_percent": psutil.cpu_percent(interval=1),
            "memory_used_gb": round(psutil.virtual_memory().used / (1024**3), 2),
            "memory_total_gb": round(psutil.virtual_memory().total / (1024**3), 2),
            "os": platform.system(),
            "uptime": self._get_uptime()
        }
        return stats

    def get_queue_stats(self):
        """تحليل طابور المهام في Redis"""
        if not self.redis:
            return {"status": "Redis disconnected"}
        
        # جلب عدد المهام المنتظرة في الطابور
        queue_length = self.redis.llen("rq:queue:omnifile_tasks") or 0
        return {
            "pending_tasks": queue_length,
            "connected_workers": self.redis.scard("rq:workers") or 0
        }

    def get_business_analytics(self):
        """إحصائيات الاستخدام والفوترة"""
        if not self.redis: return {}
        # مثال: جلب إجمالي الصفحات المعالجة هذا الشهر من جميع المستخدمين
        total_pages = 0
        keys = self.redis.keys("usage:*:2026-05") # البحث بنمط الشهر الحالي
        for key in keys:
            total_pages += int(self.redis.get(key) or 0)
        
        return {
            "total_processed_pages": total_pages,
            "active_users_this_month": len(keys)
        }

    def _get_uptime(self):
        boot_time = psutil.boot_time()
        uptime_seconds = time.time() - boot_time
        return f"{int(uptime_seconds // 3600)}h {int((uptime_seconds % 3600) // 60)}m"

admin_manager = AdminManager(os.getenv("REDIS_URL"))

```

```python
class AuthManager:
    def __init__(self):
        self.domain = AUTH0_DOMAIN
        if self.domain:
            import requests
            self.jwks_url = f"https://{self.domain}/.well-known/jwks.json"
            self.public_keys = requests.get(self.jwks_url).json()
    
    def decode_token(self, token: str) -> dict:
        import jwt
        from jwt import PyJWKClient
        if not self.domain:
            raise RuntimeError("AUTH0_DOMAIN غير معرّف في Secrets")
        jwks_client = PyJWKClient(self.jwks_url)
        signing_key = jwks_client.get_signing_key_from_jwt(token)
        return jwt.decode(token, signing_key.key, algorithms=["RS256"], audience=AUTH0_CLIENT_ID)
    
    def require_user(self, func):
        """Decorator يتحقق من JWT في كل استدعاء API"""
        @wraps(func)
        def wrapper(*args, **kwargs):
            # في Gradio نمرر الحالة يدويًا عبر gr.State
            token = kwargs.pop("auth_token", None)
            if not token:
                raise PermissionError("يجب تسجيل الدخول (أرسل token)")
            payload = self.decode_token(token)
            user_id = payload["sub"]  # e.g., 'auth0|123'
            kwargs["user_id"] = user_id
            return func(*args, **kwargs)
        return wrapper
```

```python
class UsageAnalytics:
    def __init__(self):
        self.redis = Redis.from_url(os.getenv("REDIS_URL")) if os.getenv("REDIS_URL") else None

    def get_daily_usage(self, user_id: str, days: int = 7) -> List[Dict]:
        today = datetime.utcnow().date()
        usage = []
        for i in range(days):
            day = today - timedelta(days=i)
            key = f"usage:{user_id}:{day.strftime('%Y-%m-%d')}"
            count = int(self.redis.get(key) or 0) if self.redis else 0
            usage.append({"date": day.isoformat(), "count": count})
        return sorted(usage, key=lambda x: x["date"])

    def get_total_users(self) -> int:
        if not self.redis: return 0
        # افتراض: كل مفتاح usage:user_id:* يعطي مستخدم
        keys = self.redis.keys("usage:*")
        users = set(key.split(b":")[1].decode() for key in keys)
        return len(users)
```

```python
# ==================================================================
import os, json, requests
from datetime import datetime

# ------------------------------------------------------------
# 1️⃣ التصدير المباشر إلى Overleaf (API v2)
# ------------------------------------------------------------
class OverleafExporter:
    """يرفع ملفات .tex مباشرة إلى مشروع Overleaf عبر API"""
    def __init__(self):
        self.base_url = "https://www.overleaf.com/api/v2"
        self.token = os.getenv("OVERLEAF_API_TOKEN")      # من إعدادات حساب Overleaf
        self.default_project = os.getenv("OVERLEAF_PROJECT_ID", "")  # معرف المشروع الافتراضي

    def _headers(self):
        return {"Authorization": f"Bearer {self.token}", "Content-Type": "application/json"}

    def export_to_overleaf(self, latex_content: str, filename: str = "omnifile_doc.tex", 
                          project_id: str = None) -> dict:
        """رفع ملف .tex لمشروع Overleaf"""
        if not self.token:
            return {"error": "OVERLEAF_API_TOKEN غير معرف في المتغيرات البيئية"}

        target_project = project_id or self.default_project
        if not target_project:
            return {"error": "لم يتم تحديد معرف المشروع (OVERLEAF_PROJECT_ID)"}

        # Overleaf API: PUT /projects/{id}/files/{filename}
        url = f"{self.base_url}/projects/{target_project}/files/{filename}"
        try:
            resp = requests.put(url, headers=self._headers(), data=json.dumps({"content": latex_content}))
            if resp.status_code in [200, 201]:
                return {"status": "success", "message": f"✅ تم الرفع إلى Overleaf: {filename}"}
            else:
                return {"status": "error", "details": resp.json()}
        except Exception as e:
            return {"error": str(e)}

# ------------------------------------------------------------
# 2️⃣ إشعارات Slack / Telegram
# ------------------------------------------------------------
class NotificationService:
    def __init__(self):
        self.slack_webhook = os.getenv("SLACK_WEBHOOK_URL")
        self.telegram_token = os.getenv("TELEGRAM_BOT_TOKEN")
        self.telegram_chat_id = os.getenv("TELEGRAM_CHAT_ID")

    def send_slack(self, message: str):
        if not self.slack_webhook:
            return
        requests.post(self.slack_webhook, json={"text": message})

    def send_telegram(self, message: str):
        if not self.telegram_token or not self.telegram_chat_id:
            return
        url = f"https://api.telegram.org/bot{self.telegram_token}/sendMessage"
        requests.post(url, json={"chat_id": self.telegram_chat_id, "text": message, "parse_mode": "Markdown"})

    def notify_completion(self, task_id: str, user_id: str, result_preview: str):
        """إرسال إشعار موحد عند اكتمال معالجة ملف"""
        msg = (

            f"📌 المهمة: `{task_id}`\n"
            f"👤 المستخدم: `{user_id}`\n"
            f"📄 النتيجة: {result_preview[:100]}..."
        )
        self.send_slack(msg)
        self.send_telegram(msg)

            overleaf_result = overleaf_exporter.export_to_overleaf(latex_doc, 
                                                                    filename=f"omnifile_{task_id}.tex")
            result_text += f"\n\n📤 {overleaf_result.get('message','')}"
        
        # إشعار
        notifier.notify_completion(task_id, user_id, result_text[:100])
        return result_text

print("✅ OverleafExporter + NotificationService جاهزان.")
```

## 🎁 هدية ختامية: سكريبت اختبار سريع `test_e2e.py`

```python
# test_e2e.py — اختبار نهاية-لنهاية للنظام الكامل
import requests, json, time

BASE_URL = "https://your-space.hf.space"  # استبدله برابطك الفعلي
TOKEN = "your_auth0_jwt"  # من بوابة Auth0

def test_full_pipeline():
    # 1. رفع صورة اختبار
    with open("test_sample.png", "rb") as f:
        files = {"image": f}
        r = requests.post(f"{BASE_URL}/api/ocr", files=files, 
                         headers={"Authorization": f"Bearer {TOKEN}"},
                         data={"lang": "ar"})
    task_id = r.json()["task_id"]
    
    # 2. انتظار المعالجة (Polling)
    for _ in range(30):
        r = requests.get(f"{BASE_URL}/api/status/{task_id}",
                        headers={"Authorization": f"Bearer {TOKEN}"})
        status = r.json()["status"]
        if status == "completed":
            print(f"✅ اكتملت المهمة: {r.json()['result'][:100]}...")
            break
        time.sleep(2)
    
    # 3. التحقق من المزامنة والتحليلات
    r = requests.get(f"{BASE_URL}/api/analytics", 
                    headers={"Authorization": f"Bearer {TOKEN}"})
    print(f"📊 استخدامك هذا الشهر: {r.json()['pages_processed']} صفحة")

if __name__ == "__main__":
    test_full_pipeline()
```
