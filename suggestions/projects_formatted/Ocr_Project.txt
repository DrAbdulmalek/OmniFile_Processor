# OCR Intelligent avec correction automatique et apprentissage

Cette application Streamlit permet d'extraire du texte d'images ou de PDF via plusieurs méthodes OCR (Tesseract, EasyOCR, docTR), d'afficher et comparer les résultats, d'exporter le meilleur résultat dans un fichier Word, et d'apprendre des corrections utilisateur pour s'améliorer au fil du temps.

## Fonctionnalités principales
- **OCR multi-méthodes** : Tesseract, EasyOCR, docTR (Deep Learning)
- **Comparaison automatique** : Affichage côte à côte des résultats et scores de confiance
- **Export Word** : Génération d'un fichier Word avec le texte extrait et l'image
- **Correction automatique** : Utilisation d'un modèle T5 entraîné sur vos corrections
- **Mémoire exacte** : Si une correction existe déjà pour un document, elle est utilisée directement
- **Apprentissage continu** : Ajoutez vos corrections, régénérez le dataset et réentraînez le modèle

## Installation
```bash
pip install -r requirements.txt
```

## Lancer l'application
```bash
streamlit run frontend/app.py
```

## Pipeline d'apprentissage
1. **Uploader une image/PDF** : L'OCR s'exécute, le meilleur résultat est exporté en Word
2. **Corriger le Word** : Téléchargez, corrigez, puis ré-uploadez le fichier Word corrigé
3. **Générer le dataset** :
   ```bash
   python generate_correction_csv.py
   ```
4. **Entraîner le modèle de correction** :
   ```bash
   python train_t5_correction.py
   ```
5. **Relancer l'app** : Les corrections sont désormais appliquées automatiquement

## Dépendances principales
- streamlit, opencv-python, pytesseract, easyocr, python-doctr[torch], python-docx, PyMuPDF, Pillow, torch, transformers, pandas

## Structure du projet
- `frontend/app.py` : Interface utilisateur Streamlit
- `backend/` : Scripts OCR, correction, export, prétraitement
- `corrected/` : Textes corrigés par l'utilisateur
- `correction_model/` : Modèle T5 entraîné pour la correction
- `generate_correction_csv.py` : Génère le dataset à partir des corrections
- `train_t5_correction.py` : Entraîne le modèle de correction

## Notes
- Pour de meilleurs résultats, accumulez des corrections sur vos documents récurrents.
- Le système privilégie toujours la correction humaine si elle existe pour un document donné.

import torch

# Charger le modèle et le tokenizer une seule fois
model = T5ForConditionalGeneration.from_pretrained('./correction_model')
tokenizer = T5Tokenizer.from_pretrained('./correction_model')

def correct_text(ocr_text):
    input_text = "ocr: " + ocr_text
    input_ids = tokenizer(input_text, return_tensors="pt").input_ids
    outputs = model.generate(input_ids, max_length=256)
    return tokenizer.decode(outputs[0], skip_special_tokens=True)

from docx import Document
from docx.shared import RGBColor
from docx.shared import Inches
import os

def export_to_word(text_lines, confidences, output_path, image_path=None, method=None):
    doc = Document()
    doc.add_heading('Résultat OCR', 0)

    # Si la méthode gagnante est Tesseract, on insère l'image originale
    if method == "tesseract" and image_path is not None and os.path.exists(image_path):
        doc.add_picture(image_path, width=Inches(5.5))
        doc.add_paragraph("\nTexte extrait :\n")

    for line, conf in zip(text_lines, confidences):
        p = doc.add_paragraph()
        run = p.add_run(line)
        if conf < 85:
            run.font.color.rgb = RGBColor(255, 255, 0)  # Jaune
    doc.save(output_path)

import os
os.environ["KMP_DUPLICATE_LIB_OK"] = "TRUE"
os.environ["CUDA_VISIBLE_DEVICES"] = ""
os.environ["TF_CPP_MIN_LOG_LEVEL"] = "3"
import warnings
warnings.filterwarnings("ignore")
from backend.preprocessing import preprocess_image
from backend.ocr_tesseract import ocr_tesseract
from backend.ocr_easyocr import ocr_easyocr
from backend.ocr_doctr import ocr_doctr
from backend.export import export_to_word
from backend.corrector import correct_text

def run_all_ocr_methods(image_paths):
    results = {"tesseract": {"lines": [], "confs": []},
               "easyocr": {"lines": [], "confs": []},
               "doctr": {"lines": [], "confs": []}}

    # Mémoire exacte : si une correction existe déjà pour la première image, on l'utilise directement
    base_name = os.path.splitext(os.path.basename(image_paths[0]))[0]
    corrected_path = os.path.join("corrected", base_name + "_corrige.txt")
    if os.path.exists(corrected_path):
        with open(corrected_path, "r", encoding="utf-8") as f:
            corrected_lines = f.read().splitlines()
        # On remplit les résultats avec la correction pour toutes les méthodes (pour l'affichage)
        for method in results:
            results[method]["lines"] = corrected_lines
            results[method]["confs"] = [100] * len(corrected_lines)
            results[method]["avg_conf"] = 100
        best_method = "doctr"  # ou autre, ici arbitraire car c'est la correction qui compte
        os.makedirs("output", exist_ok=True)
        output_path = os.path.join("output", f"result_{best_method}.docx")
        export_to_word(corrected_lines, [100]*len(corrected_lines), output_path, image_path=image_paths[0], method=best_method)
        return results, best_method, output_path

    # Sinon, pipeline normal OCR + correction automatique
    for image_path in image_paths:
        image = preprocess_image(image_path)
        lines_tess, confs_tess = ocr_tesseract(image, return_conf=True)
        results["tesseract"]["lines"].extend(lines_tess)
        results["tesseract"]["confs"].extend(confs_tess)

        lines_easy, confs_easy = ocr_easyocr(image_path)
        results["easyocr"]["lines"].extend(lines_easy)
        results["easyocr"]["confs"].extend(confs_easy)

        lines_doctr, confs_doctr = ocr_doctr(image_path)
        results["doctr"]["lines"].extend(lines_doctr)
        results["doctr"]["confs"].extend(confs_doctr)

    for method in results:
        confs = results[method]["confs"]
        results[method]["avg_conf"] = sum(confs) / len(confs) if confs else 0

    best_method = max(results.items(), key=lambda x: x[1]["avg_conf"])[0]
    os.makedirs("output", exist_ok=True)
    output_path = os.path.join("output", f"result_{best_method}.docx")
    export_to_word(results[best_method]["lines"], results[best_method]["confs"], output_path, image_path=image_paths[0], method=best_method)

    # Correction automatique après OCR
    ocr_text = "\n".join(results[best_method]["lines"])
    corrected_text = correct_text(ocr_text)
    results[best_method]["lines"] = corrected_text.split("\n")

    return results, best_method, output_path

from doctr.io import DocumentFile
from doctr.models import ocr_predictor

def ocr_doctr(image_path):
    doc = DocumentFile.from_images(image_path)
    model = ocr_predictor(pretrained=True)
    result = model(doc)
    export = result.export()
    lines = []
    for block in export["pages"][0]["blocks"]:
        for line in block["lines"]:
            line_text = " ".join([word["value"] for word in line["words"]])
            lines.append(line_text)
    confs = [100] * len(lines)
    return lines, confs

def ocr_easyocr(path):
    results = reader.readtext(path)
    lines = []
    confidences = []
    for detection in results:
        lines.append(detection[1])
        confidences.append(detection[2] * 100)
    return lines, confidences

from pytesseract import Output

def ocr_tesseract(image, return_conf=False):
    data = pytesseract.image_to_data(image, lang='fra', config='--psm 6', output_type=Output.DICT)
    lines = []
    confidences = []
    current_line = ""
    current_conf = []
    last_line_num = -1

    for i, word in enumerate(data['text']):
        if word.strip() == "":
            continue
        line_num = data['line_num'][i]
        if line_num != last_line_num and current_line:
            lines.append(current_line.strip())
            confidences.append(sum(current_conf)/len(current_conf) if current_conf else 100)
            current_line = word + " "
            current_conf = [data['conf'][i]]
            last_line_num = line_num
        else:
            current_line += word + " "
            current_conf.append(data['conf'][i])
            last_line_num = line_num
    if current_line:
        lines.append(current_line.strip())
        confidences.append(sum(current_conf)/len(current_conf) if current_conf else 100)
    if return_conf:
        return lines, confidences
    return "\n".join(lines)

from transformers import TrOCRProcessor, VisionEncoderDecoderModel
from PIL import Image
import torch

processor = TrOCRProcessor.from_pretrained("microsoft/trocr-base-handwritten")
model = VisionEncoderDecoderModel.from_pretrained("microsoft/trocr-base-handwritten")

def ocr_trocr(path):
    image = Image.open(path).convert("RGB")
    pixel_values = processor(images=image, return_tensors="pt").pixel_values
    generated_ids = model.generate(pixel_values)
    text = processor.batch_decode(generated_ids, skip_special_tokens=True)[0]
    return [text], [90]  # supposition: texte complet avec une confiance de 90%

import cv2
import numpy as np

def preprocess_image(path):
    image = cv2.imread(path)
    if image is None:
        raise FileNotFoundError(f"Image introuvable : {path}")
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    coords = cv2.findNonZero(cv2.bitwise_not(gray))
    angle = cv2.minAreaRect(coords)[-1]
    angle = -(90 + angle) if angle < -45 else -angle
    (h, w) = gray.shape[:2]
    M = cv2.getRotationMatrix2D((w // 2, h // 2), angle, 1.0)
    gray = cv2.warpAffine(gray, M, (w, h), flags=cv2.INTER_CUBIC, borderMode=cv2.BORDER_REPLICATE)
    binary = cv2.adaptiveThreshold(gray, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
                                   cv2.THRESH_BINARY, 11, 2)
    binary = cv2.medianBlur(binary, 3)
    kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (2, 2))
    return cv2.morphologyEx(binary, cv2.MORPH_OPEN, kernel)

import os
os.environ["KMP_DUPLICATE_LIB_OK"] = "TRUE"
import glob
import pandas as pd
from backend.ocr_tesseract import ocr_tesseract
from backend.ocr_easyocr import ocr_easyocr
from backend.ocr_doctr import ocr_doctr
from backend.preprocessing import preprocess_image

corrected_dir = "corrected"
images_dir = "images"
csv_path = "corrections.csv"

data = []

fichiers_corriges = glob.glob(os.path.join(corrected_dir, "*.txt"))

for txt_file in fichiers_corriges:
    base_name = os.path.basename(txt_file).replace("_corrige.txt", "")
    # On cherche l'image d'origine
    possible_images = glob.glob(os.path.join(images_dir, base_name + ".*"))
    if not possible_images:
        continue
    image_path = possible_images[0]
    # OCR brut (ici Tesseract, tu peux changer pour EasyOCR ou docTR)
    image = preprocess_image(image_path)
    ocr_lines, _ = ocr_tesseract(image, return_conf=True)
    ocr_text = "\n".join(ocr_lines)
    # Texte corrigé
    with open(txt_file, "r", encoding="utf-8") as f:
        corrige_text = f.read()
    data.append({"ocr": ocr_text, "corrige": corrige_text})

# Sauvegarde en CSV
if data:
    df = pd.DataFrame(data)
    df.to_csv(csv_path, index=False, encoding="utf-8")
else:
    pass

import pandas as pd
from transformers import T5ForConditionalGeneration, T5Tokenizer, Seq2SeqTrainer, Seq2SeqTrainingArguments
from torch.utils.data import Dataset
import torch

# 1. Charger le dataset
print('Chargement du dataset...')
df = pd.read_csv('corrections.csv')

# 2. Préparer les entrées pour T5
inputs = ["ocr: " + str(o) for o in df['ocr']]
targets = [str(c) for c in df['corrige']]

# 3. Tokenizer et modèle
print('Chargement du modèle et du tokenizer...')
tokenizer = T5Tokenizer.from_pretrained('t5-small')
model = T5ForConditionalGeneration.from_pretrained('t5-small')

# 4. Dataset PyTorch
class CorrectionDataset(Dataset):
    def __init__(self, inputs, targets, tokenizer, max_length=256):
        self.inputs = inputs
        self.targets = targets
        self.tokenizer = tokenizer
        self.max_length = max_length

    def __len__(self):
        return len(self.inputs)

    def __getitem__(self, idx):
        input_enc = self.tokenizer(self.inputs[idx], truncation=True, padding='max_length', max_length=self.max_length, return_tensors="pt")
        target_enc = self.tokenizer(self.targets[idx], truncation=True, padding='max_length', max_length=self.max_length, return_tensors="pt")
        return {
            'input_ids': input_enc['input_ids'].squeeze(),
            'attention_mask': input_enc['attention_mask'].squeeze(),
            'labels': target_enc['input_ids'].squeeze()
        }

print('Préparation du dataset...')
train_dataset = CorrectionDataset(inputs, targets, tokenizer)

# 5. Entraînement
print('Début de l\'entraînement...')
training_args = Seq2SeqTrainingArguments(
    output_dir='./correction_model',
    num_train_epochs=5,
    per_device_train_batch_size=4,
    save_steps=1000,
    save_total_limit=2,
    logging_steps=100
)

trainer = Seq2SeqTrainer(
    model=model,
    args=training_args,
    train_dataset=train_dataset
)

trainer.train()
model.save_pretrained('./correction_model')
tokenizer.save_pretrained('./correction_model')
print("Modèle de correction entraîné et sauvegardé dans ./correction_model")
